sc_clean <- startup_combined %>%
  mutate(
    # Extract all 4-digit numbers (years) from the string
    years = str_extract_all(years_of_operation, "\\d{4}"),
    # Extract start year as the first year found
    start_year = as.numeric(sapply(years, `[`, 1)),
    # Extract end year as the second year found (if any)
    end_year = as.numeric(sapply(years, `[`, 2)),
    # If end_year is NA (only one year found), set it equal to start_year
    end_year = ifelse(is.na(end_year), start_year, end_year),
    # Calculate year length
    year_length = end_year - start_year
  ) %>%
  select(-years)  # Remove intermediate column

Startups in this dataset have varied lifespans. Understanding the average and range of operation durations helps us gauge typical startup longevity.

library(dplyr)

duration_summary <- sc_clean %>%
  summarise(
    count = n(),
    mean_duration = mean(year_length, na.rm = TRUE),
    median_duration = median(year_length, na.rm = TRUE),
    min_duration = min(year_length, na.rm = TRUE),
    max_duration = max(year_length, na.rm = TRUE),
    sd_duration = sd(year_length, na.rm = TRUE)
  )

Startup Endings and Potential Crisis Impact Understanding when startups end is crucial, especially to see if external shocks like financial crises increase failure rates.

end_year_counts <- sc_clean %>%
  count(end_year) %>%
  arrange(end_year)

crisis_years <- c(2008, 2020)

end_year_counts <- end_year_counts %>%
  mutate(crisis = ifelse(end_year %in% crisis_years, "Crisis Year", "Other Year"))

ggplot(end_year_counts, aes(x = end_year, y = n, fill = crisis)) +
  geom_col() +
  scale_fill_manual(values = c("Crisis Year" = "red", "Other Year" = "grey")) +
  labs(title = "Number of Startups Ended per Year with Crisis Years Highlighted",
       x = "End Year",
       y = "Count of Startups",
       fill = "") +
  theme_minimal()

The highlighted bars show that startup failures tend to spike around the 2008 and 2020 crisis years, suggesting external economic shocks may significantly impact startup survival.
